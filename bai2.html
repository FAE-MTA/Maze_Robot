Bài 2. Điều khiển các loại LED

Bài học giúp học viên hiểu các loại LED phổ biến, cách đấu nối và lập trình điều khiển LED từ LED thường đến LED NeoPixel vòng 16 LED trên robot.

1. Mục tiêu bài học

  • Hiểu nguyên lý hoạt động của LED thường, LED RGB, LED NeoPixel và LED NeoPixel Ring.
  • Biết cách đấu nối từng loại LED với Arduino Mega 2560.
  • Biết sử dụng pinMode(), digitalWrite(), analogWrite().
  • Biết điều khiển LED vòng WS2812 gồm 16 LED trên robot.
  • Ứng dụng LED để báo trạng thái robot: sẵn sàng, đang chạy, lỗi.

2. Tổng quan các loại LED

Trong robot và các hệ thống nhúng, LED thường được dùng để hiển thị trạng thái hoạt động, báo lỗi, báo chế độ chạy hoặc tạo hiệu ứng trực quan khi robot vận hành.

3. LED thường

3.1. Nguyên lý

LED thường là loại LED đơn sắc, chỉ có hai trạng thái cơ bản: sáng hoặc tắt. Khi Arduino xuất mức HIGH ra chân điều khiển, LED sáng. Khi xuất mức LOW, LED tắt.

3.2. Đấu dây

  • Chân dương LED → điện trở 220Ω → chân D8 Arduino.
  • Chân âm LED → GND.
LED thường cần mắc nối tiếp với điện trở hạn dòng để tránh làm hỏng LED hoặc chân Arduino.

3.3. Lệnh điều khiển

pinMode(8, OUTPUT);        // Đặt chân D8 ở chế độ xuất tín hiệu
digitalWrite(8, HIGH);     // Xuất mức 5V → LED sáng
digitalWrite(8, LOW);      // Xuất mức 0V → LED tắt

3.4. Code ví dụ

#define LED_PIN 8          // LED thường nối với chân D8

void setup() {
  pinMode(LED_PIN, OUTPUT);        // Đặt chân D8 là OUTPUT
}

void loop() {
  digitalWrite(LED_PIN, HIGH);     // Bật LED
  delay(1000);                     // Chờ 1 giây

  digitalWrite(LED_PIN, LOW);      // Tắt LED
  delay(1000);                     // Chờ 1 giây
}

4. LED RGB

4.1. Nguyên lý

LED RGB gồm ba LED nhỏ bên trong: đỏ, xanh lá và xanh dương. Bằng cách thay đổi độ sáng của từng màu, ta có thể tạo ra nhiều màu khác nhau.

4.2. Đấu dây

  • Chân R → chân PWM D7.
  • Chân G → chân PWM D6.
  • Chân B → chân PWM D5.
  • Chân chung → GND hoặc 5V tùy loại LED RGB.
Các chân PWM trên Arduino Mega 2560: 2 → 13 và 44 → 46.

4.3. Lệnh điều khiển

pinMode(7, OUTPUT);        // Đặt chân D7 điều khiển màu đỏ
pinMode(6, OUTPUT);        // Đặt chân D6 điều khiển màu xanh lá
pinMode(5, OUTPUT);        // Đặt chân D5 điều khiển màu xanh dương

analogWrite(7, 255);       // Đỏ sáng mạnh nhất
analogWrite(6, 0);         // Tắt xanh lá
analogWrite(5, 0);         // Tắt xanh dương

4.4. Code ví dụ

int R = 7;     // Chân PWM điều khiển LED đỏ
int G = 6;     // Chân PWM điều khiển LED xanh lá
int B = 5;     // Chân PWM điều khiển LED xanh dương

void setup() {
  pinMode(R, OUTPUT);      // Đặt chân R là OUTPUT
  pinMode(G, OUTPUT);      // Đặt chân G là OUTPUT
  pinMode(B, OUTPUT);      // Đặt chân B là OUTPUT
}

void loop() {
  analogWrite(R, 255);     // Đỏ sáng mạnh nhất
  analogWrite(G, 0);       // Tắt xanh lá
  analogWrite(B, 0);       // Tắt xanh dương
  delay(1000);

  analogWrite(R, 0);       // Tắt đỏ
  analogWrite(G, 255);     // Xanh lá sáng mạnh nhất
  analogWrite(B, 0);       // Tắt xanh dương
  delay(1000);

  analogWrite(R, 0);       // Tắt đỏ
  analogWrite(G, 0);       // Tắt xanh lá
  analogWrite(B, 255);     // Xanh dương sáng mạnh nhất
  delay(1000);
}

5. LED NeoPixel đơn

5.1. Nguyên lý

LED NeoPixel là LED RGB thông minh, thường dùng chip WS2812. Khác với LED RGB thường, NeoPixel chỉ cần một chân DATA để điều khiển màu. Nếu nối nhiều NeoPixel nối tiếp nhau, chỉ cần một chân DATA duy nhất để điều khiển toàn bộ chuỗi LED. Chân DOUT của LED trước sẽ nối vào chân DIN của LED sau.

5.2. Đấu dây

  • DATA → D19.
  • VCC → 5V.
  • GND → GND.

5.3. Lệnh điều khiển

led.setPixelColor(0, led.Color(255, 0, 0));   // Gán màu đỏ cho LED số 0
led.show();                                   // Cập nhật màu ra LED thật

5.4. Code ví dụ

#include <Adafruit_NeoPixel.h>

#define LED_PIN 19       // Chân DATA của LED NeoPixel đơn nối D19
#define NUM_LEDS 1       // Số lượng LED NeoPixel đơn

Adafruit_NeoPixel led(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

void setup() {
  led.begin();           // Khởi tạo LED NeoPixel
  led.show();            // Tắt LED ban đầu
}

void loop() {
  led.setPixelColor(0, led.Color(255, 0, 0));  // LED số 0 màu đỏ
  led.show();                                  // Cập nhật LED
  delay(1000);

  led.setPixelColor(0, led.Color(0, 255, 0));  // LED số 0 màu xanh lá
  led.show();                                  // Cập nhật LED
  delay(1000);

  led.setPixelColor(0, led.Color(0, 0, 255));  // LED số 0 màu xanh dương
  led.show();                                  // Cập nhật LED
  delay(1000);
}

6. LED NeoPixel Ring trên robot

6.1. Nguyên lý

LED NeoPixel Ring là vòng gồm nhiều LED WS2812 nối tiếp nhau. Robot sử dụng vòng LED gồm 16 LED. Mỗi LED có thể điều khiển màu độc lập. Nếu nối nhiều NeoPixel Ring nối tiếp nhau, chân DOUT của LED trước sẽ nối vào chân DIN của LED sau.

6.2. Đấu dây

  • DATA → D13.
  • VCC → 5V.
  • GND → GND.
Cần nối chung GND giữa Arduino và LED Ring. Nếu không nối chung GND, LED có thể không hoạt động đúng.

6.3. Lệnh điều khiển

ring.setPixelColor(i, ring.Color(0, 0, 255)); // Gán màu xanh dương cho LED thứ i
ring.show();                                  // Cập nhật toàn bộ vòng LED

6.4. Code ví dụ cơ bản

#include <Adafruit_NeoPixel.h>

#define LED_PIN 13       // Chân DATA của LED Ring nối D13
#define NUM_LEDS 16      // LED Ring trên robot có 16 LED

Adafruit_NeoPixel ring(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

void setup() {
  ring.begin();          // Khởi tạo LED Ring
  ring.show();           // Tắt toàn bộ LED ban đầu
}

void loop() {
  for (int i = 0; i < NUM_LEDS; i++) {
    ring.setPixelColor(i, ring.Color(0, 0, 255)); // LED thứ i màu xanh dương
  }

  ring.show();           // Cập nhật màu ra LED Ring
}
#include <Adafruit_NeoPixel.h>

#define LED_PIN 13
#define NUM_LEDS 16

Adafruit_NeoPixel ring(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

int speedDelay = 40;    // Thời gian trễ, càng nhỏ chạy càng nhanh
int colorShift = 0;     // Biến dịch màu

uint32_t wheel(byte pos) {
  pos = 255 - pos;

  if (pos < 85) {
    return ring.Color(255 - pos * 3, 0, pos * 3);
  }

  if (pos < 170) {
    pos -= 85;
    return ring.Color(0, pos * 3, 255 - pos * 3);
  }

  pos -= 170;
  return ring.Color(pos * 3, 255 - pos * 3, 0);
}

void fadeAll(byte amount) {
  for (int i = 0; i < NUM_LEDS; i++) {
    uint32_t c = ring.getPixelColor(i);

    uint8_t r = (c >> 16) & 0xFF;
    uint8_t g = (c >> 8) & 0xFF;
    uint8_t b = c & 0xFF;

    r = (r * amount) / 255;
    g = (g * amount) / 255;
    b = (b * amount) / 255;

    ring.setPixelColor(i, r, g, b);
  }
}

void setup() {
  ring.begin();          // Khởi tạo LED Ring
  ring.show();           // Tắt LED ban đầu
}

void loop() {
  for (int i = 0; i < NUM_LEDS; i++) {
    fadeAll(180);        // Làm mờ các LED cũ để tạo đuôi sáng

    int colorIndex = (i * 16 + colorShift) % 256;

    ring.setPixelColor(i, wheel(colorIndex)); // Gán màu đổi dần cho LED thứ i
    ring.show();                              // Cập nhật LED Ring

    delay(speedDelay);
  }

  colorShift++;          // Dịch màu sau mỗi vòng chạy
}

#include <Adafruit_NeoPixel.h>

// ===== LED thường =====
#define LED_THUONG 8   // D8

// ===== LED RGB =====
#define R 7   // đỏ
#define G 6   // xanh lá
#define B 5   // xanh dương

// ===== NeoPixel đơn =====
#define LED_SINGLE_PIN 19
#define NUM_SINGLE 1

Adafruit_NeoPixel ledSingle(NUM_SINGLE, LED_SINGLE_PIN, NEO_GRB + NEO_KHZ800);

// ===== NeoPixel vòng =====
#define LED_RING_PIN 13
#define NUM_RING 16

Adafruit_NeoPixel ring(NUM_RING, LED_RING_PIN, NEO_GRB + NEO_KHZ800);


// ===== HÀM PHỤ =====
void setRingColor(int r, int g, int b) {
  for (int i = 0; i < NUM_RING; i++) {
    ring.setPixelColor(i, ring.Color(r, g, b));
  }
  ring.show();
}

void setup() {

  // LED thường
  pinMode(LED_THUONG, OUTPUT);   // D8 OUTPUT

  // LED RGB
  pinMode(R, OUTPUT);   // D7 đỏ
  pinMode(G, OUTPUT);   // D6 xanh lá
  pinMode(B, OUTPUT);   // D5 xanh dương

  // NeoPixel
  ledSingle.begin();    // LED đơn
  ledSingle.show();

  ring.begin();         // LED vòng
  ring.show();
}

void loop() {

  // ===== TEST LED THƯỜNG =====
  digitalWrite(LED_THUONG, HIGH);  // Bật LED thường
  delay(500);
  digitalWrite(LED_THUONG, LOW);   // Tắt LED thường


  // ===== TEST LED RGB =====
  analogWrite(R, 255);  // Đỏ
  analogWrite(G, 0);
  analogWrite(B, 0);
  delay(500);

  analogWrite(R, 0);
  analogWrite(G, 255);  // Xanh lá
  analogWrite(B, 0);
  delay(500);

  analogWrite(R, 0);
  analogWrite(G, 0);
  analogWrite(B, 255);  // Xanh dương
  delay(500);


  // ===== TEST NEOPIXEL ĐƠN =====
  ledSingle.setPixelColor(0, ledSingle.Color(255, 255, 0)); // Vàng
  ledSingle.show();
  delay(500);


  // ===== TEST LED VÒNG =====
  setRingColor(0, 0, 255);  // Xanh dương
  delay(500);

  setRingColor(255, 0, 0);  // Đỏ
  delay(500);

}

6.5. Ứng dụng trong robot

  • Xanh lá: robot sẵn sàng.
  • Xanh dương: robot đang chạy.
  • Đỏ: robot gặp lỗi hoặc cần kiểm tra.
  • Hiệu ứng chạy vòng: robot đang xử lý hoặc đang chờ lệnh.

7. Bài tập thực hành

  • Thay đổi thời gian nhấp nháy của LED thường.
  • Tạo màu vàng bằng LED RGB.
  • Điều khiển NeoPixel đơn lần lượt hiển thị đỏ, xanh lá, xanh dương.
  • Làm LED Ring sáng từng LED từ vị trí 0 đến 15.
  • Thiết kế 3 màu trạng thái cho robot: sẵn sàng, đang chạy, lỗi.

8. Lỗi thường gặp

Lỗi 1: LED thường không sáng do đấu ngược cực hoặc thiếu điện trở.
Lỗi 2: LED RGB không đổi độ sáng do nối vào chân không hỗ trợ PWM.
Lỗi 3: NeoPixel không chạy do thiếu thư viện Adafruit NeoPixel.
Lỗi 4: LED Ring sai màu do chọn sai chuẩn màu NEO_GRB.
Lỗi 5: LED Ring chập chờn do nguồn yếu hoặc chưa nối chung GND.
